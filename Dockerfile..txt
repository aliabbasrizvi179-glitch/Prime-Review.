# Use PHP-FPM with Nginx
FROM php:8.2-fpm-bullseye as php

# Install system dependencies
RUN apt-get update && apt-get install -y \
    git \
    unzip \
    libzip-dev \
    && docker-php-ext-install zip

# Install Composer
COPY --from=composer:2.6 /usr/bin/composer /usr/bin/composer

# Set working directory
WORKDIR /var/www/html

# Copy application files
COPY . .

# Install PHP dependencies
RUN composer install --no-dev --optimize-autoloader --no-interaction

# Set proper permissions
RUN chmod 755 /var/www/html && \
    touch users.json error.log && \
    chmod 644 users.json && \
    chmod 666 error.log && \
    chown -R www-data:www-data /var/www/html

# Nginx stage
FROM nginx:1.23-alpine as webserver

# Copy custom nginx config
COPY ./web-server-config/nginx.conf /etc/nginx/conf.d/default.conf

# Copy PHP files from the PHP stage
COPY --from=php /var/www/html /var/www/html

# Expose port (Render will use PORT environment variable)
EXPOSE 8080

# Start nginx
CMD ["nginx", "-g", "daemon off;"]
